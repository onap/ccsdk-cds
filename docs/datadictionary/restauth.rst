Resource Rest Authentication 
----------------------------

token-auth:
~~~~~~~~~~~

.. code-block:: json
   :linenos:
   
   {
      "dsl_definitions": {
        "dynamic-rest-source": {
          "type" : "token-auth",
          "url" : "http://localhost:32778",
          "token" : "<token>"
        }
      }
   }
   
basic-auth:
~~~~~~~~~~~

.. code-block:: json
   :linenos:
      
   {
      "dsl_definitions": {
        "dynamic-rest-source": {
          "type" : "basic-auth",
          "url" : "http://localhost:32778",
          "username" : "<username>",
          "password": "<password>"
       }
      }
   }
   
ssl-basic-auth:
~~~~~~~~~~~~~~~

.. code-block:: json
   :linenos:
      
   {
      "dsl_definitions": {
        "dynamic-rest-source": {
          "type" : "ssl-basic-auth",
          "url" : "http://localhost:32778",
          "keyStoreInstance": "JKS or PKCS12",
          "sslTrust": "trusture",
          "sslTrustPassword": "<password>",
          "sslKey": "keystore",
          "sslKeyPassword": "<password>"
       }
      }
   }
